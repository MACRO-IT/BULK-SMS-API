Get started

    API Endpoint

        https://kwachacredit.com/api/api_testing

The kwachacredit API provides programmatic for payments collections from customers. With our APIs, you can make and receive payments in real world from your clients online. You don't know how to work with API's? Don't worry we have composed our API's into beautiful HTML standard code snippets in various languages which you can easily copy and paste in your code and start collecting payments.

To use this in real world, you need a Bearer Token. Please create an account with us and make your token in the developers dashboard.

NSMutableURLRequest *request = [NSMutableURLRequest requestWithURL:[NSURL URLWithString:@"https://kwachacredit.com/api/api_testing"]
  cachePolicy:NSURLRequestUseProtocolCachePolicy
  timeoutInterval:10.0];

  @"Content-Type": @"application/json",
  @"Accept": @"application/json",
  @"Authorization": @"Bearer ......"
};

[request setAllHTTPHeaderFields:headers];
NSData *postData = [[NSData alloc] initWithData:[@"{\r\n  \"amount\": \"300\",\r\n  \"currency\": \"EUR\",\r\n  \"externalId\": \"279182\",\r\n  \"payer\": {\r\n    \"partyIdType\": \"MSISDN\",\r\n    \"partyId\": \"9612572923\"\r\n  },\r\n  \"payerMessage\": \"Paying For Chickens\",\r\n  \"payeeNote\": \"Payment Successfull\"\r\n}" dataUsingEncoding:NSUTF8StringEncoding]];
[request setHTTPBody:postData];

[request setHTTPMethod:@"GET"];

NSURLSession *session = [NSURLSession sharedSession];
NSURLSessionDataTask *dataTask = [session dataTaskWithRequest:request
completionHandler:^(NSData *data, NSURLResponse *response, NSError *error) {
  if (error) {
    NSLog(@"%@", error);
    dispatch_semaphore_signal(sema);
  } else {
    NSHTTPURLResponse *httpResponse = (NSHTTPURLResponse *) response;
    NSError *parseError = nil;
    NSDictionary *responseDictionary = [NSJSONSerialization JSONObjectWithData:data options:0 error:&parseError];
    NSLog(@"%@",responseDictionary);
    dispatch_semaphore_signal(sema);
  }
}];
[dataTask resume];
dispatch_semaphore_wait(sema, DISPATCH_TIME_FOREVER);
